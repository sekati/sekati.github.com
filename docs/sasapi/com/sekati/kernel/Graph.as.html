    10  import com.sekati.display.BaseClip;
    11  import com.sekati.draw.Rectangle;
    12  import com.sekati.draw.Line;
    13  import com.sekati.except.IllegalArgumentException;
    14  import com.sekati.geom.Point;
    15  import com.sekati.utils.ClassUtils;
    16  
    17  /**
    18   * Graph
    19   */
    20  class com.sekati.kernel.Graph {
    21  
    22  	private static var W:Number = 600;
    23  	private static var H:Number = 60;
    24  	private static var HDIV:Number = 3;
    25  	private static var HOFFSET:Number = 15;
    26  	private static var WDIV:Number = 10;
    27  	private static var WOFFSET:Number = 60;
    28  	private static var FILL:Number = 0xFFFFFF;
    29  	private static var BORDER:Number = 0x000000;
    30  	private static var HLCOLOR:Number = 0xEDEDED;
    31  	private static var WLCOLOR:Number = 0xEDEDED;
    32  	private var _this:Graph;	
    33  	private var graph:MovieClip;
    34  	
    35  	/**
    36  	 * Constructor
    37  	 * @param target (MovieClip)
    38  	 * @return Void
    39  	 */
    40  	public function Graph(target:MovieClip) {
    41  		if (!target) {
    42  			throw new IllegalArgumentException(this, "Graph expects 'target' parameter.", arguments);	
    43  		}
    44  		_this = this;
    45  		graph = ClassUtils.createEmptyMovieClip(com.sekati.display.BaseClip, target, "graph", {_x:0, _y:0} );
    46  		Rectangle.draw(graph, new Point(0,0), new Point(W, H), FILL, 100, 1, BORDER, 100);
    47  		// wlines
    48  		for (var j:Number = 0; j < WDIV; j++) {
    49  			var w:Number = int(WOFFSET*j)+WOFFSET;
    50  			var l1:MovieClip = graph.createEmptyMovieClip("lw"+j, graph.getNextHighestDepth());
    51  			new Line(l1, new Point(w,1), new Point(w,H-1), 1, WLCOLOR, 100);				
    52  		}		
    53  		// hlines
    54  		for (var i:Number = 0; i < HDIV; i++) {
    55  			var h:Number = int(HOFFSET*i)+HOFFSET;
    56  			var l0:MovieClip = graph.createEmptyMovieClip("lh"+i, graph.getNextHighestDepth());
    57  			new Line(l0, new Point(1,h), new Point(W,h), 1, HLCOLOR, 100);
    58  		}
    59  	}
    60  }
