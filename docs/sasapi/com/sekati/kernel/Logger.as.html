    17  /**
    18   * Logger is a multi-tiered debugging tool designed to clarify the debugging process.
    19   * 
    20   * The Logger consists of two main tiered components:
    21   * 	- "levels" (method calls, i.e. 'trace', 'info', 'status', etc).
    22   * 	- "filters" (origin of the level call, i.e. '_root', 'myClassInstance', 'myMc', etc).
    23   * 	
    24   * New level method calls may be created dynamically "overloading" the singleton class 
    25   * via {@link setLevel}, __resolve & proxy as illustrated in the exam	ple usage below.
    26   * NOTE: custom levels/overloading requires publishing >= Flash 9.
    27   * 
    28   * Logger can be configured to display its output to the  {@link Console} or Flash IDE 
    29   * Output Panel via:
    30   * {@code Logger.$.ide = true;
    31   * 	Logger.$.console = true;
    32   * }
    33   * 
    34   * {@code Usage:
    35   * 	Logger.$.trace( this, "Hello World!" );
    36   * 	Logger.$.setLevel( "sekati", true );
    37   * 	Logger.$.sekati( this, "I created a new level!" );
    38   * 	Logger.$.setLevel("sekati", false );
    39   * 	Logger.$.sekati( this, "on noes; you wont see this cuz the level is disabled!" );
    40   * 	Logger.$.setFilter ( this, true );
    41   * 	Logger.$.status ( this, "oh noes; you wont see this cuz the origin is filtered!" );
    42   * }
    43   * 
    44   * @see {@link com.sekati.log.Console}
    45   */

   243  	/**
   244  	 * object is a special level (method) which handles object recursion via {@link com.sekati.log.Inspector}

   255  	
   256  	/**
   257  	 * Enable/disable a filter and create the filter if it does not already exist.
   258  	 * @param origin (Object) object to filter on [usually a string]
   259  	 * @param isFiltered (Boolean)
   260  	 * @return Void
   261  	 */
   262  	public function setFilter(origin:Object, isFiltered:Boolean):Void {
   263  		if (isFiltered) {
   264  			filter( origin );
   265  		} else {
   266  			unfilter( origin );
   267  		}
   268  	}

   276  	}

   278  	/**
   279  	 * Reset all filters (clearing previous filters)
   280  	 * @return Void
   281  	 */
   282  	public function resetFilters():Void {
   283  		_filters = [];
   284  	}
   285  	
   286  	/**
   287  	 * Check if an origin's output is being filtered
   288  	 * @param origin (Object) to check.
   289  	 * @return Boolean
   290  	 */
   291  	public function isFiltered(origin:Object):Boolean {
   292  		var o:String = String( origin );
   293  		for (var i:Number = 0; i < _filters.length; i++) {
   294  			if (_filters[i] == o) {
   295  				return true;
   296  			}
   297  		}
   298  		return false;
   299  	}	
   300  	
   301  	/**
   302  	 * Add an origin to the filters array.
   303  	 * @param origin (Object) to be added.
   304  	 * @return Void
   305  	 */
   306  	private function filter(origin:Object):Void {
   307  		var o:String = String( origin );
   308  		if (!isFiltered( o )) {
   309  			_filters.push ( o );
   310  		}
   311  	}
   312  
   313  	/**
   314  	 * Remove an origin from the filters array.
   315  	 * @param origin (Object) to be removed.
   316  	 * @return Void
   317  	 */
   318  	private function unfilter(origin:Object):Void {
   319  		var o:String = String( origin );
   320  		for (var i:Number = 0; i < _filters.length; i++) {
   321  			if (_filters[i] == o) {
   322  				_filters.splice( i, 1 );
   323  				break;
   324  			}
   325  		}
   326  	}

   328  	// OUTPUT HANDLERS
   329  	
   330  	/**
   331  	 * Manges output to {@link Console} and Flash Output Panel.
   332  	 * @param level (String)
   333  	 * @param origin (Object)
   334  	 * @param msg (Object)
   335  	 * @return Void
   336  	 */
   337  	private function _output(level:String, origin:Object, msg:Object):Void {
   338  		// validate that we should be outputting this content: Logger enabled, level enabled, origin unfiltered & proper LogEvent.
   339  		if (_logEnable == false || _levels[level] == false || isFiltered( origin ) == true || !(level.toLowerCase().indexOf("__get__") <= -1)) {
   340  			return;
   341  		}
   342  		var benchmark:Number = _watch.lap();
   343  		var id:Number = _logId++;
   344  		var e:LogEvent = new LogEvent( {id:id, type:level.toLowerCase(), origin:String(origin), message:String(msg), benchmark:benchmark} );
   345  		
   346  		// dispatch event to localConnection Console
   347  		if(console) {
   348  			LCBinding.send( e );
   349  		}	
   350  
   351  		if(ide) {
   352  			trace( id+"\t"+level.toUpperCase()+"\t"+origin+"\t"+msg+"\t("+benchmark+" ms)" );
   353  		}
   354  	}

   356  	// LEVEL OVERLOADING	
   357  
   358  	/**
   359  	 * __resolve catches all wrapper & manually created level methods and processes them thru
   360  	 * the proxy and sends to _output; super cool!
   361  	 * @param name (String)
   362  	 * @return Function
   363  	 */
   364  	private function __resolve(name:String):Function {
   365  		if (name.indexOf( LogEvent.LOG_EVENT ) > 1) {
   366  			trace("!!!!!!!!! logger returning: "+name+" reflects: "+Stringifier.stringify(LogEvent));
   367  			return;
   368  		}
   369  		var f:Function = function():Object {
   370  			arguments.unshift( name );
   371  			return __proxy.apply( _proxyObj, arguments );
   372  		};
   373  		_proxyObj[name] = f;
   374  		return f;
   375  	}
   376  
   377  	/**
   378  	 * Reconstruct methods calls and rereoute to output
   379  	 * @param name (String)
   380  	 * @return Void
   381  	 */
   382  	private function __proxy(name:String):Void {
   383  		arguments.shift ();
   384  		var n:String = String( name ).toLowerCase ();
   385  		var o:String = String( arguments[0] );
   386  		var s:String = String( arguments[1] );
   387  		_instance._output( n, o, s );
   388  	}

   390  	/**
   391  	 * Destroy the Singleton instance.
   392  	 * @return Void
   393  	 */
   394  	public function destroy():Void {
   395  		_watch.destroy();
   396  		LCBinding.disconnect();
   397  		for(var i in _instance) {
   398  			delete _instance[i];	
   399  		}
   400  		delete _instance;
   401  	}
